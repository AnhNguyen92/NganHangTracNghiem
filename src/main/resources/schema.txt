#create database
create database `bank_question`;

CREATE TABLE `bank_question`.`users` ( `id` BIGINT NOT NULL AUTO_INCREMENT , `username` VARCHAR(50) NOT NULL , `password` VARCHAR(100) NOT NULL , `first_name` VARCHAR(50) NULL , `last_name` INT(100) NULL , `create_date` DATE NOT NULL , `email` VARCHAR(100) NULL , `phone_number` VARCHAR(20) NULL , `role` ENUM('ADMIN','SUPERVISOR','INSPECTOR','USER') NOT NULL , `status` ENUM('ACTIVE','IN_ACTIVE') NOT NULL , `avatar_file` VARCHAR(100) NULL , `date_of_birth` DATE NULL , `gender` ENUM('MALE','FEMALE') NOT NULL , PRIMARY KEY (`id`), UNIQUE `u_username` (`username`)) ENGINE = InnoDB;
